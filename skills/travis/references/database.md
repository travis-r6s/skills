# Database Patterns

## Drizzle ORM (Cloudflare D1 / SQLite)

Used in Nuxt apps deployed to Cloudflare Workers via NuxtHub (`@nuxthub/core`).

### Schema Definition

```ts
// server/database/schema.ts
import { relations } from 'drizzle-orm'
import { index, sqliteTable, text, uniqueIndex } from 'drizzle-orm/sqlite-core'

export const users = sqliteTable('users', {
  id: text('id').primaryKey().notNull(),
  name: text('name').notNull(),
  email: text('email').notNull(),
}, table => ({
  emailIndex: uniqueIndex('users_email_index').on(table.email),
}))

export const posts = sqliteTable('posts', {
  id: text('id').primaryKey().notNull(),
  userId: text('user_id').notNull(),
  title: text('title').notNull(),
  body: text('body'),
}, (table) => ({
  userIdIndex: index('posts_user_id_idx').on(table.userId),
}))

export const usersRelations = relations(users, ({ many }) => ({
  posts: many(posts),
}))

export const postsRelations = relations(posts, ({ one }) => ({
  user: one(users, { fields: [posts.userId], references: [users.id] }),
}))
```

### DB Utility (NuxtHub pattern)

```ts
// server/utils/db.ts
import type { InferInsertModel, InferSelectModel } from 'drizzle-orm'
import type { SQLiteTableWithColumns } from 'drizzle-orm/sqlite-core'
import { drizzle } from 'drizzle-orm/d1'
import { createInsertSchema, createSelectSchema } from 'drizzle-zod'
import * as schema from '../database/schema'

export { and, eq, or, sql } from 'drizzle-orm'

// Filter out relation definitions from schema
type SchemaTableNames = {
  [K in keyof typeof schema]: (typeof schema)[K] extends SQLiteTableWithColumns<any> ? K : never
}[keyof typeof schema]

/** Get the SELECT type for a table. */
export type Doc<T extends SchemaTableNames> = InferSelectModel<(typeof schema)[T]>

/** Get the INSERT type for a table. */
export type DocInsert<T extends SchemaTableNames> = InferInsertModel<(typeof schema)[T]>

export const tables = schema

/** Get a Drizzle DB instance backed by the NuxtHub D1 database. */
export function useDB() {
  return drizzle(hubDatabase(), { schema })
}
```

### Querying

```ts
// In Nuxt server routes
const db = useDB()

// Select
const user = await db.select().from(tables.users).where(eq(tables.users.id, userId)).get()

// Insert
await db.insert(tables.posts).values({ id: ulid(), userId, title, body })

// With relations (requires schema passed to drizzle())
const userWithPosts = await db.query.users.findFirst({
  where: eq(tables.users.id, userId),
  with: { posts: true },
})
```

### Zod Schemas from Drizzle

```ts
// drizzle-zod generates Zod schemas from table definitions
const insertUserSchema = createInsertSchema(tables.users).omit({ id: true })
const selectUserSchema = createSelectSchema(tables.users)

// Use with h3-zod for request validation
```

### drizzle-kit Config

```ts
// drizzle.config.ts
import { defineConfig } from 'drizzle-kit'

export default defineConfig({
  dialect: 'sqlite',
  schema: './server/database/schema.ts',
  out: './server/database/migrations',
})
```

Scripts:
```json
{
  "db:generate": "drizzle-kit generate",
  "db:migrate": "drizzle-kit migrate"
}
```

## Kysely (PostgreSQL / Xata)

Used in Node/Nuxt projects with PostgreSQL or Xata serverless databases. Type-safe SQL query builder.

```ts
import { Kysely, PostgresDialect, CamelCasePlugin } from 'kysely'
import { Pool } from 'pg'
import type { DB } from './schema'  // generated by kysely-codegen

const db = new Kysely<DB>({
  dialect: new PostgresDialect({ pool: new Pool({ connectionString: process.env.DATABASE_URL }) }),
  plugins: [new CamelCasePlugin()],  // converts snake_case columns to camelCase
})

export default db
```

Generate types from your PostgreSQL schema:
```bash
kysely-codegen --dialect postgres --out-file src/db/schema.d.ts
```

Querying:
```ts
// Select with join
const users = await db
  .selectFrom('users')
  .innerJoin('posts', 'posts.userId', 'users.id')
  .select(['users.id', 'users.email', 'posts.title'])
  .where('users.id', '=', userId)
  .execute()

// Insert
await db.insertInto('users').values({ id: ulid(), email, name }).execute()

// Update
await db.updateTable('users').set({ name }).where('id', '=', userId).execute()
```

With Xata:
```ts
import { XataDialect } from '@xata.io/kysely'
import { getXataClient } from './xata'

const db = new Kysely<DB>({ dialect: new XataDialect({ xata: getXataClient() }) })
```

## Lucid ORM (AdonisJS)

See [architecture.md](architecture.md) for full AdonisJS model patterns.

Quick reference:
```ts
// app/Models/User.ts
import { BaseModel, column, hasMany } from '@ioc:Adonis/Lucid/Orm'  // v5
// OR for v6:
import { BaseModel, column, hasMany } from '@adonisjs/lucid/orm'

export default class User extends BaseModel {
  @column({ isPrimary: true }) public id: number
  @column() public email: string
  @column.dateTime({ autoCreate: true }) public createdAt: DateTime
  @column.dateTime({ autoCreate: true, autoUpdate: true }) public updatedAt: DateTime

  @hasMany(() => Post)
  public posts: HasMany<typeof Post>
}

// Query
const user = await User.find(id)
const users = await User.query().where('email', email).firstOrFail()
await User.create({ email, name })
```
